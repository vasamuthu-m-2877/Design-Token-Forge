<button class="btn" :data-variant="variant ?? 'filled'" :data-role="role ?? 'brand'" :data-size="size ?? 'base'" :data-rounded="rounded || undefined" :data-loading="loading || undefined" :disabled="disabled" > <span class="btn__label"><slot /></span> </button>
